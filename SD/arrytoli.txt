void push(STACK**s,char item)
	{

	 STACK *temp,*r;
	 if(*s==NULL)
	 {
		temp=(STACK*)malloc(sizeof(STACK));
		temp->data=item;
		temp->link=NULL;
		*s=temp;
	 }
	 else
	 {
		temp=*s;
		while(temp->link!=NULL)
			temp=temp->link;
		r=(STACK*)malloc(sizeof(STACK));
		r->data=item;
		r->link=NULL;
		temp->link=r;
	 }
}
void display(STACK*q)
{
	while(q!=NULL)
	{
		printf("%c",q->data);
		q=q->link;
	}
}
char pop(STACK**x)
	{
		char c;
		STACK *ptr;
		ptr=*x;
		c=(*x)->data;
		*x=(*x)->link;
		return c;
		free(ptr);
	}
